 45 > map <- data.frame(code      = c(19, 21, 22),
 46 +                   hospital  = c(1L, 1L, 1L),
 47 +                   inpatient = c(0L, 1L, 0L))

 49 > # Another map that could be built to cover all the codes in the data set:
 50 > map_df  <- data.frame(code = unique(psps_2019_df$PLACE_OF_SERVICE_CD))
 51 > map_tbl <- tibble(code     = unique(psps_2019_df$PLACE_OF_SERVICE_CD))

 53 >
 54 > map_df$hospital  <- as.integer(map_df$code %in% c(19, 21, 22))

 75 +   base_v1 = {

 87 +   }
 88 +   ,

 90 +     # A left merge
 91 +     base::merge(

113 +              )
114 +   }
115 +   ,
116 +   tidy_merge = {
117 +     dplyr::left_join(
118 +                        x = psps_2019_tidy
119 +                      , y = map_tbl
120 +                      , by = c("PLACE_OF_SERVICE_CD" = "code")
121 +     )
122 +   }
123 +   ,
124 +   dt_v1 = {
125 +     psps_2019_dt[, `:=`(hospital = as.integer(PLACE_OF_SERVICE_CD %in% c(19, 21, 22)), inpatient = NA_integer_)]

127 +   }
128 +   ,
129 +   dt_v2 = {
130 +     psps_2019_dt[, hospital := as.integer(PLACE_OF_SERVICE_CD %in% c(19, 21, 22))]

132 +   }
133 +   ,
134 +   dt_merge = {
